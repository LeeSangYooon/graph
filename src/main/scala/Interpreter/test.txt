precision = 500
mandelbrot(x, y, n) =
    if (n == 0) {precision}
    else {
        if (x^2 + y^2 > 4) {
            n
        } else {
            mandelbrot(x, y, n - 1)
        }
    }

color(x, y) = {
    mandelbrot(x, y, precision)
}
graph = [color, x: -2 ~ 2, y: -2 ~ 2]
show graph


derivative(f) = x => (limit h -> 0: (f(x + h) - f(x)) / h)
rep(f, a, n) = if (n > 0) rep(f, f(a), n-1) else a
rep(derivative, x => x^11, 10)


f(x) = x ^ 2 + 2 * x + 1
